Keskustelupalstan tulee tarjota ominaisuudet keskustelualueiden luomiseen, 
keskustelunavausten l‰hett‰miseen keskustelualueelle, sek‰ tietysti viesteihin vastaamiseen. 
Viestien l‰hett‰j‰t tulee pysty‰ tunnistamaan esimerkiksi viesti‰ kirjoitettaessa annettavan nimimerkin avulla. 

- keskustelupalsta (ei tarvetta m‰‰ritell‰)
- keskustelualue (= Alue)
- keskusteluavaus (= Aihe)
- viesti (= Viesti)
- l‰hett‰j‰ (= Nimimerkki)
- nimimerkki (= sama kuin l‰hett‰j‰)

Viesti
- (pk) id int
- (fk) nimimerkki : Nimimerkki
- (fk) aihe : Aihe
- aikaleima date
- teksti varchar(100)

Nimimerkki
- (pk) kayttajatunnus varchar(20)

Aihe
- (pk) id int
- (fk) alue : Alue
- teksti varchar(40)

Alue
- (pk) id int
- otsikko varchar(40)

Create Table -lauseet:

CREAT TABLE Viesti 
(
	id int PRIMARY KEY, 
	nimimerkki varchar(20), 
	aihe int, 
	aikaleima date, 
	teksti varchar(100), 
	FOREIGN KEY(nimimerkki) REFERENCES Nimimerkki(kayttajatunnus),
	FOREIGN KEY(aihe) REFERENCES Aihe(id)
);

CREAT TABLE Nimimerkki
(
	kayttajatunnus varchar(20) PRIMARY KEY
);

CREATE TABLE Aihe
(
	id int PRIMARY KEY,
	alue int,
	teksti varchar(40),
	FOREIGN KEY(alue) REFERENCES Alue(id)
);

CREATE TABLE Alue
(
	id int PRIMARY KEY,
	otsikko varchar(40)
);
